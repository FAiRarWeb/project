// Move button event listeners outside the component, after the component registration document.addEventListener('DOMContentLoaded', function() { const animateButton = document.querySelector("#animateButton"); const infoButton = document.querySelector("#infoButton"); const googleButton = document.querySelector("#googleButton"); const videoEl = document.querySelector("#video1"); let isAnimatePressed = false; // Animate Me button - controls video playback animateButton.addEventListener("click", () => { const activeTarget = document.querySelector("[mindar-image-target].active"); if (!activeTarget) return; const videoPlane = activeTarget.querySelector("a-video"); // Toggle pressed state for color change animateButton.classList.toggle("pressed"); // Toggle video playback isAnimatePressed = !isAnimatePressed; // Hide the "i" button when video is played if (isAnimatePressed) { googleButton.style.display = "none"; animateButton.classList.add("pressed"); videoPlane.setAttribute("visible", true); videoEl.play().catch(e => console.warn("Autoplay blocked:", e)); } else { animateButton.classList.remove("pressed"); videoPlane.setAttribute("visible", false); videoEl.pause(); } console.log("Animate button pressed, isAnimatePressed:", isAnimatePressed); }); // Info button - controls "i" button visibility infoButton.addEventListener("click", () => { console.log("Info button clicked"); // Debug log const activeTarget = document.querySelector("[mindar-image-target].active"); console.log("Active target:", activeTarget); // Debug log if (!activeTarget) { console.log("No active target found"); return; } const videoPlane = activeTarget.querySelector("a-video"); // Toggle info button visibility const isHidden = googleButton.style.display === "none" || googleButton.style.display === ""; console.log("Google button is hidden:", isHidden); // Debug log if (isHidden) { infoButton.classList.add("pressed"); googleButton.style.display = "block"; console.log("Showing Google button"); // Close video when info button is pressed videoPlane.setAttribute("visible", false); videoEl.pause(); isAnimatePressed = false; animateButton.classList.remove("pressed"); } else { infoButton.classList.remove("pressed"); googleButton.style.display = "none"; console.log("Hiding Google button"); } });
